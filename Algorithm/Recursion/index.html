function factorial_recursion(n){
    if(n === 1) return 1; //base case
    return n * factorial_recursion(n-1);
}

//for loop
function factorial_loop (n) {
  var result = 1;
  for (var i = n; i >= 1; i--) {
    result *= i;
  }
  return result;
}
